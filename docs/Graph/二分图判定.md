按照和起点的距离的奇偶性分类，黑白染色，若相同奇偶性不同色，则不是二分图

```c++
int col[N];
vector<int> ed[N];
bool judgeBfs(int r)
{
    col[r]=1;
    queue<int> d;
    d.push(r);
    int u;
    while(!d.empty())
    {
        u=d.front();
        d.pop();
        for(int v:ed[u])
        {
            if(col[v]==0)
            {
                col[v]=-col[u];
                d.push(v);
            }
            else if(col[v]==col[u])
                return false;
        }
    }
    return true;
}
bool judge()
{
    memset(col,0,sizeof(col));
    for(int i=1;i<=n;++i)
    {
        if(!col[i])
        {
            if(!judgeBfs(i))
                return false;
        }
    }
    return true;
}
```
